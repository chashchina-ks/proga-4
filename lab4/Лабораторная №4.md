"""
Паттерн "Состояние" (State)
Позволяет объекту менять свое поведение при изменении внутреннего состояния.
"""

from abc import ABC, abstractmethod
from typing import Optional

# Альтернативный пример: Автомат с напитками (более простой)
class VendingMachineState(ABC):
    """Состояние автомата с напитками"""
    
    def __init__(self, machine: 'VendingMachine'):
        self._machine = machine
    
    @abstractmethod
    def insert_coin(self) -> None:
        pass
    
    @abstractmethod
    def select_drink(self) -> None:
        pass
    
    @abstractmethod
    def dispense(self) -> None:
        pass


class NoCoinState(VendingMachineState):
    def insert_coin(self) -> None:
        print("Монета принята. Выберите напиток.")
        self._machine.set_state(HasCoinState(self._machine))
    
    def select_drink(self) -> None:
        print("Сначала бросьте монету!")
    
    def dispense(self) -> None:
        print("Бросьте монету и выберите напиток.")


class HasCoinState(VendingMachineState):
    def insert_coin(self) -> None:
        print("Монета уже есть. Выберите напиток.")
    
    def select_drink(self) -> None:
        print("Напиток выбран. Получите свой напиток!")
        self._machine.set_state(DispensingState(self._machine))
    
    def dispense(self) -> None:
        print("Сначала выберите напиток.")


class DispensingState(VendingMachineState):
    def insert_coin(self) -> None:
        print("Подождите, выдача напитка...")
    
    def select_drink(self) -> None:
        print("Уже выдается напиток...")
    
    def dispense(self) -> None:
        print("Ваш напиток выдан. Спасибо!")
        self._machine.set_state(NoCoinState(self._machine))


class VendingMachine:
    def __init__(self):
        self._state = NoCoinState(self)
    
    def set_state(self, state):
        self._state = state
    
    def insert_coin(self):
        self._state.insert_coin()
    
    def select_drink(self):
        self._state.select_drink()
    
    def dispense(self):
        self._state.dispense()


def demo_vending_machine():
    print("\n" + "=" * 50)
    print("Альтернативный пример: Торговый автомат")
    print("=" * 50)
    
    machine = VendingMachine()
    
    machine.select_drink()  # Ошибка: нет монеты
    machine.insert_coin()   # Ок: монета принята
    machine.insert_coin()   # Ошибка: монета уже есть
    machine.select_drink()  # Ок: выдача напитка
    machine.dispense()      # Ок: получение напитка
    machine.dispense()      # Ошибка: нужно снова бросить монету


if __name__ == "__main__":
    main()
    demo_vending_machine()
